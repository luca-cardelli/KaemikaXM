//======================================
// Positive Test Signal - Sine
//======================================

// Turn a positive function into a waveform
network Signal(species X, function f) {
   Ø ->{{100*f()}} X;   X ->{100} Ø
}

species A @ 0 M
Signal(A, λ(){1+sin(time)})

report A            // plot test signal
report 1+sin(time)  // compare with true function
equilibrate for 10
